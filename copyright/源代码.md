   130  class _PyiArgumentParser(argparse.ArgumentParser):
   131      def __init__(self, *args, **kwargs):
   132          self._pyi_action_groups = defaultdict(list)
   133          super().__init__(*args, **kwargs)
   134  
   135      def _add_options(self, __add_options: callable, name: str = ""):
   136          """
   137          Mutate self with the given callable, storing any new actions added in a named group
   138          """
   139          n_actions_before = len(getattr(self, "_actions", []))
   140          __add_options(self)  # preserves old behavior
   141          new_actions = getattr(self, "_actions", [])[n_actions_before:]
   142          self._pyi_action_groups[name].extend(new_actions)
   143  
   144      def _option_name(self, action):
   145          """
   146          Get the option name(s) associated with an action
   147  
   148          For options that define both short and long names, this function will
   149          return the long names joined by "/"
   150          """
   151          longnames = [name for name in action.option_strings if name.startswith("--")]
   152          if longnames:
   153              name = "/".join(longnames)
   154          else:
   155              name = action.option_strings[0]
   156          return name
   157  
   158      def _forbid_options(self, args: argparse.Namespace, group: str, errmsg: str = ""):
   159          """Forbid options from a named action group"""
   160          options = defaultdict(str)
   161          for action in self._pyi_action_groups[group]:
   162              dest = action.dest
   163              name = self._option_name(action)
   164              if getattr(args, dest) is not self.get_default(dest):
   165                  if dest in options:
   166                      options[dest] += "/"
   167                  options[dest] += name
   168  
   169          # if any options from the forbidden group are not the default values,
   170          # the user must have passed them in, so issue an error report
   171          if options:
   172              sep = "\n  "
   173              bad = sep.join(options.values())
   174              if errmsg:
   175                  errmsg = "\n" + errmsg
   176              raise SystemExit(f"ERROR: option(s) not allowed:{sep}{bad}{errmsg}")

   207  def run(pyi_args: list | None = None, pyi_config: dict | None = None):
   208      """
   209      pyi_args     allows running PyInstaller programmatically without a subprocess
   210      pyi_config   allows checking configuration once when running multiple tests
   211      """
   212      compat.check_requirements()
   213      check_unsafe_privileges()
   214  
   215      import PyInstaller.log
   216  
   217      old_sys_argv = sys.argv
   218      try:
   219          parser = generate_parser()
   220          autocomplete(parser)
   221          if pyi_args is None:
   222              pyi_args = sys.argv[1:]
   223          try:
   224              index = pyi_args.index("--")
   225          except ValueError:
   226              index = len(pyi_args)
   227          args = parser.parse_args(pyi_args[:index])
   228          spec_args = pyi_args[index + 1:]
   229          PyInstaller.log.__process_options(parser, args)
   230  
   231          # Print PyInstaller version, Python version, and platform as the first line to stdout. This helps us identify
   232          # PyInstaller, Python, and platform version when users report issues.
   233          try:
   234              from _pyinstaller_hooks_contrib import __version__ as contrib_hooks_version
   235          except Exception:
   236              contrib_hooks_version = 'unknown'
   237  
   238          logger.info('PyInstaller: %s, contrib hooks: %s', __version__, contrib_hooks_version)
   239          logger.info('Python: %s%s', platform.python_version(), " (conda)" if compat.is_conda else "")
   240          logger.info('Platform: %s', platform.platform())
   241          logger.info('Python environment: %s', sys.prefix)
   242  
   243          # Skip creating .spec when .spec file is supplied.
   244          if args.filenames[0].endswith('.spec'):
   245              parser._forbid_options(
   246                  args, group="makespec", errmsg="makespec options not valid when a .spec file is given"
   247              )
   248              spec_file = args.filenames[0]
   249          else:
   250              # Ensure that the given script files exist, before trying to generate the .spec file.
   251              # This prevents us from overwriting an existing (and customized) .spec file if user makes a typo in the
   252              # .spec file's suffix when trying to  build it, for example, `pyinstaller program.cpes` (see #8276).
   253              # It also prevents creation of a .spec file when `pyinstaller program.py` is accidentally ran from a
   254              # directory that does not contain the script (for example, due to failing to change the directory prior
   255              # to running the command).
   256              for filename in args.filenames:
   257                  if not os.path.isfile(filename):
   258                      raise SystemExit(f"ERROR: Script file {filename!r} does not exist.")
   259              spec_file = run_makespec(**vars(args))
   260  
   261          sys.argv = [spec_file, *spec_args]
   262          run_build(pyi_config, spec_file, **vars(args))
   263  
   264      except KeyboardInterrupt:
   265          raise SystemExit("Aborted by user request.")
   266      except RecursionError:
   267          from PyInstaller import _recursion_too_deep_message
   268          _recursion_too_deep_message.raise_with_msg()
   269      finally:
   270          sys.argv = old_sys_argv

   281  def check_unsafe_privileges():
   282      """
   283      Forbid dangerous usage of PyInstaller with escalated privileges
   284      """
   285      if compat.is_win and not compat.is_win_wine:
   286          # Discourage (with the intention to eventually block) people using *run as admin* with PyInstaller.
   287          # There are 4 cases, block case 3 but be careful not to also block case 2.
   288          #   1. User has no admin access: TokenElevationTypeDefault
   289          #   2. User is an admin/UAC disabled (common on CI/VMs): TokenElevationTypeDefault
   290          #   3. User has used *run as administrator* to elevate: TokenElevationTypeFull
   291          #   4. User can escalate but hasn't: TokenElevationTypeLimited
   292          # https://techcommunity.microsoft.com/t5/windows-blog-archive/how-to-determine-if-a-user-is-a-member-of-the-administrators/ba-p/228476
   293          import ctypes
   294  
   295          advapi32 = ctypes.CDLL("Advapi32.dll")
   296          kernel32 = ctypes.CDLL("kernel32.dll")
   297  
   298          kernel32.GetCurrentProcess.restype = ctypes.c_void_p
   299          process = kernel32.GetCurrentProcess()
   300  
   301          token = ctypes.c_void_p()
   302          try:
   303              TOKEN_QUERY = 8
   304              assert advapi32.OpenProcessToken(ctypes.c_void_p(process), TOKEN_QUERY, ctypes.byref(token))
   305  
   306              elevation_type = ctypes.c_int()
   307              TokenElevationType = 18
   308              assert advapi32.GetTokenInformation(
   309                  token, TokenElevationType, ctypes.byref(elevation_type), ctypes.sizeof(elevation_type),
   310                  ctypes.byref(ctypes.c_int())
   311              )
   312          finally:
   313              kernel32.CloseHandle(token)
   314  
   315          if elevation_type.value == 2:  # TokenElevationTypeFull
   316              logger.log(
   317                  logging.DEPRECATION,
   318                  "Running PyInstaller as admin is not necessary nor sensible. Run PyInstaller from a non-administrator "
   319                  "terminal. PyInstaller 7.0 will block this."
   320              )
   321  
   322      elif compat.is_darwin or compat.is_linux:
   323          # Discourage (with the intention to eventually block) people using *sudo* with PyInstaller.
   324          # Again there are 4 cases, block only case 4.
   325          #   1. Non-root: os.getuid() != 0
   326          #   2. Logged in as root (usually a VM): os.getlogin() == "root", os.getuid() == 0
   327          #   3. No named users (e.g. most Docker containers): os.getlogin() fails
   328          #   4. Regular user using escalation: os.getlogin() != "root", os.getuid() == 0
   329          try:
   330              user = os.getlogin()
   331          except OSError:
   332              user = ""
   333          if os.getuid() == 0 and user and user != "root":
   334              logger.log(
   335                  logging.DEPRECATION,
   336                  "Running PyInstaller as root is not necessary nor sensible. Do not use PyInstaller with sudo. "
   337                  "PyInstaller 7.0 will block this."
   338              )
   339  
   340      if compat.is_win:
   341          # Do not let people run PyInstaller from admin cmd's default working directory (C:\Windows\system32)
   342          cwd = pathlib.Path.cwd()
   343  
   344          try:
   345              win_dir = compat.win32api.GetWindowsDirectory()
   346          except Exception:
   347              win_dir = None
   348          win_dir = None if win_dir is None else pathlib.Path(win_dir).resolve()
   349  
   350          inside_win_dir = cwd == win_dir or win_dir in cwd.parents
   351  
   352          # The only exception to the above is if user's home directory is also located under %WINDIR%, which happens
   353          # when PyInstaller is ran under SYSTEM user.
   354          if inside_win_dir:
   355              home_dir = pathlib.Path.home().resolve()
   356              if cwd == home_dir or home_dir in cwd.parents:
   357                  inside_win_dir = False
   358  
   359          if inside_win_dir:
   360              raise SystemExit(
   361                  f"ERROR: Do not run pyinstaller from {cwd}. cd to where your code is and run pyinstaller from there. "
   362                  "Hint: You can open a terminal where your code is by going to the parent folder in Windows file "
   363                  "explorer and typing cmd into the address bar."
   364              )
   365  
   366  
   367  if __name__ == '__main__':
   368      run()

   440  def _pyi_machine(machine, system):
   441      # type: (str, str) -> str
   442      """
   443      Choose an intentionally simplified architecture identifier to be used in the bootloader's directory name.
   444  
   445      Args:
   446          machine:
   447              The output of ``platform.machine()`` or any known architecture alias or shorthand that may be used by a
   448              C compiler.
   449          system:
   450              The output of ``platform.system()`` on the target machine.
   451      Returns:
   452          Either a string tag or, on platforms that don't need an architecture tag, ``None``.
   453  
   454      Ideally, we would just use ``platform.machine()`` directly, but that makes cross-compiling the bootloader almost
   455      impossible, because you need to know at compile time exactly what ``platform.machine()`` will be at run time, based
   456      only on the machine name alias or shorthand reported by the C compiler at the build time. Rather, use a loose
   457      differentiation, and trust that anyone mixing armv6l with armv6h knows what they are doing.
   458      """
   459      # See the corresponding tests in tests/unit/test_compat.py for examples.
   460  
   461      if platform.machine() == "sw_64" or platform.machine() == "loongarch64":
   462          # This explicitly inhibits cross compiling the bootloader for or on SunWay and LoongArch machine.
   463          return platform.machine()
   464  
   465      if system == "Windows":
   466          if machine.lower().startswith("arm"):
   467              return "arm"
   468          else:
   469              return "intel"
   470  
   471      if system == "SunOS":
   472          if machine.lower() in ("x86", "i86pc"):
   473              return "intel"
   474          else:
   475              return "sparc"
   476  
   477      if system != "Linux":
   478          # No architecture specifier for anything par Linux.
   479          # - macOS is on two 64 bit architectures, but they are merged into one "universal2" bootloader.
   480          # - BSD supports a wide range of architectures, but according to PyPI's download statistics, every one of our
   481          #   BSD users are on x86_64. This may change in the distant future.
   482          return
   483  
   484      if machine.startswith(("arm", "aarch")):
   485          # ARM has a huge number of similar and aliased sub-versions, such as armv5, armv6l armv8h, aarch64.
   486          return "arm"
   487      if machine in ("thumb"):
   488          # Reported by waf/gcc when Thumb instruction set is enabled on 32-bit ARM. The platform.machine() returns "arm"
   489          # regardless of the instruction set.
   490          return "arm"
   491      if machine in ("x86_64", "x64", "x86"):
   492          return "intel"
   493      if re.fullmatch("i[1-6]86", machine):
   494          return "intel"
   495      if machine.startswith(("ppc", "powerpc")):
   496          # PowerPC comes in 64 vs 32 bit and little vs big endian variants.
   497          return "ppc"
   498      if machine in ("mips64", "mips"):
   499          return "mips"
   500      if machine.startswith("riscv"):
   501          return "riscv"
   502      # Machines with no known aliases :)
   503      if machine in ("s390x",):
   504          return machine
   505  
   506      # Unknown architectures are allowed by default, but will all be placed under one directory. In theory, trying to
   507      # have multiple unknown architectures in one copy of PyInstaller will not work, but that should be sufficiently
   508      # unlikely to ever happen.
   509      return "unknown"

   570  class CArchiveReader:
   571      """
   572      Reader for PyInstaller's CArchive (PKG) archive.
   573      """
   574  
   575      # Cookie - holds some information for the bootloader. C struct format definition. '!' at the beginning means network
   576      # byte order. C struct looks like:
   577      #
   578      # typedef struct _archive_cookie
   579      # {
   580      #     char magic[8];
   581      #     uint32_t pkg_length;
   582      #     uint32_t toc_offset;
   583      #     uint32_t toc_length;
   584      #     uint32_t python_version;
   585      #     char python_libname[64];
   586      # } ARCHIVE_COOKIE;
   587      #
   588      _COOKIE_MAGIC_PATTERN = b'MEI\014\013\012\013\016'
   589  
   590      _COOKIE_FORMAT = '!8sIIII64s'
   591      _COOKIE_LENGTH = struct.calcsize(_COOKIE_FORMAT)
   592  
   593      # TOC entry:
   594      #
   595      # typedef struct _toc_entry
   596      # {
   597      #     uint32_t entry_length;
   598      #     uint32_t offset;
   599      #     uint32_t length;
   600      #     uint32_t uncompressed_length;
   601      #     unsigned char compression_flag;
   602      #     char typecode;
   603      #     char name[1]; /* Variable-length name, padded to multiple of 16 */
   604      # } TOC_ENTRY;
   605      #
   606      _TOC_ENTRY_FORMAT = '!IIIIBc'
   607      _TOC_ENTRY_LENGTH = struct.calcsize(_TOC_ENTRY_FORMAT)
   608  
   609      def __init__(self, filename):
   610          self._filename = filename
   611          self._start_offset = 0
   612          self._end_offset = 0
   613          self._toc_offset = 0
   614          self._toc_length = 0
   615  
   616          self.toc = {}
   617          self.options = []
   618  
   619          # Load TOC
   620          with open(self._filename, "rb") as fp:
   621              # Find cookie MAGIC pattern
   622              cookie_start_offset = self._find_magic_pattern(fp, self._COOKIE_MAGIC_PATTERN)
   623              if cookie_start_offset == -1:
   624                  raise ArchiveReadError("Could not find COOKIE magic pattern!")
   625  
   626              # Read the whole cookie
   627              fp.seek(cookie_start_offset, os.SEEK_SET)
   628              cookie_data = fp.read(self._COOKIE_LENGTH)
   629  
   630              magic, archive_length, toc_offset, toc_length, pyvers, pylib_name = \
   631                  struct.unpack(self._COOKIE_FORMAT, cookie_data)
   632  
   633              # Compute start and end offset of the the archive
   634              self._end_offset = cookie_start_offset + self._COOKIE_LENGTH
   635              self._start_offset = self._end_offset - archive_length
   636  
   637              # Verify that Python shared library name is set
   638              if not pylib_name:
   639                  raise ArchiveReadError("Python shared library name not set in the archive!")
   640  
   641              # Read whole toc
   642              fp.seek(self._start_offset + toc_offset)
   643              toc_data = fp.read(toc_length)
   644  
   645              self.toc, self.options = self._parse_toc(toc_data)
   646  
   647      @staticmethod
   648      def _find_magic_pattern(fp, magic_pattern):
   649          # Start at the end of file, and scan back-to-start
   650          fp.seek(0, os.SEEK_END)
   651          end_pos = fp.tell()
   652  
   653          # Scan from back
   654          SEARCH_CHUNK_SIZE = 8192
   655          magic_offset = -1
   656          while end_pos >= len(magic_pattern):
   657              start_pos = max(end_pos - SEARCH_CHUNK_SIZE, 0)
   658              chunk_size = end_pos - start_pos
   659              # Is the remaining chunk large enough to hold the pattern?
   660              if chunk_size < len(magic_pattern):
   661                  break
   662              # Read and scan the chunk
   663              fp.seek(start_pos, os.SEEK_SET)
   664              buf = fp.read(chunk_size)
   665              pos = buf.rfind(magic_pattern)
   666              if pos != -1:
   667                  magic_offset = start_pos + pos
   668                  break
   669              # Adjust search location for next chunk; ensure proper overlap
   670              end_pos = start_pos + len(magic_pattern) - 1
   671  
   672          return magic_offset
   673  
   674      @classmethod
   675      def _parse_toc(cls, data):
   676          options = []
   677          toc = {}
   678          cur_pos = 0
   679          while cur_pos < len(data):
   680              # Read and parse the fixed-size TOC entry header
   681              entry_length, entry_offset, data_length, uncompressed_length, compression_flag, typecode = \
   682                  struct.unpack(cls._TOC_ENTRY_FORMAT, data[cur_pos:(cur_pos + cls._TOC_ENTRY_LENGTH)])
   683              cur_pos += cls._TOC_ENTRY_LENGTH
   684              # Read variable-length name
   685              name_length = entry_length - cls._TOC_ENTRY_LENGTH
   686              name, *_ = struct.unpack(f'{name_length}s', data[cur_pos:(cur_pos + name_length)])
   687              cur_pos += name_length
   688              # Name string may contain up to 15 bytes of padding
   689              name = name.rstrip(b'\0').decode('utf-8')
   690  
   691              typecode = typecode.decode('ascii')
   692  
   693              # The TOC should not contain duplicates, except for OPTION entries. Therefore, keep those
   694              # in a separate list. With options, the rest of the entries do not make sense, anyway.
   695              if typecode == 'o':
   696                  options.append(name)
   697              else:
   698                  toc[name] = (entry_offset, data_length, uncompressed_length, compression_flag, typecode)
   699  
   700          return toc, options
   701  
   702      def extract(self, name):
   703          """
   704          Extract data for the given entry name.
   705          """
   706  
   707          entry = self.toc.get(name)
   708          if entry is None:
   709              raise KeyError(f"No entry named {name!r} found in the archive!")
   710  
   711          entry_offset, data_length, uncompressed_length, compression_flag, typecode = entry
   712          with open(self._filename, "rb") as fp:
   713              fp.seek(self._start_offset + entry_offset, os.SEEK_SET)
   714              data = fp.read(data_length)
   715  
   716          if compression_flag:
   717              import zlib
   718              data = zlib.decompress(data)
   719  
   720          return data
   721  
   722      def raw_pkg_data(self):
   723          """
   724          Extract complete PKG/CArchive archive from the parent file (executable).
   725          """
   726          total_length = self._end_offset - self._start_offset
   727          with open(self._filename, "rb") as fp:
   728              fp.seek(self._start_offset, os.SEEK_SET)
   729              return fp.read(total_length)
   730  
   731      def open_embedded_archive(self, name):
   732          """
   733          Open new archive reader for the embedded archive.
   734          """
   735  
   736          entry = self.toc.get(name)
   737          if entry is None:
   738              raise KeyError(f"No entry named {name!r} found in the archive!")
   739  
   740          entry_offset, data_length, uncompressed_length, compression_flag, typecode = entry
   741  
   742          if typecode == PKG_ITEM_PYZ:
   743              # Open as embedded archive, without extraction.
   744              return ZlibArchiveReader(self._filename, self._start_offset + entry_offset)
   745          elif typecode == PKG_ITEM_ZIPFILE:
   746              raise NotAnArchiveError("Zipfile archives not supported yet!")
   747          else:
   748              raise NotAnArchiveError(f"Entry {name!r} is not a supported embedded archive!")
   749  
   750  
   751  def pkg_archive_contents(filename, recursive=True):
   752      """
   753      List the contents of the PKG / CArchive. If `recursive` flag is set (the default), the contents of the embedded PYZ
   754      archive is included as well.
   755  
   756      Used by the tests.
   757      """
   758  
   759      contents = []
   760  
   761      pkg_archive = CArchiveReader(filename)
   762      for name, toc_entry in pkg_archive.toc.items():
   763          *_, typecode = toc_entry
   764          contents.append(name)
   765          if typecode == PKG_ITEM_PYZ and recursive:
   766              pyz_archive = pkg_archive.open_embedded_archive(name)
   767              for name in pyz_archive.toc.keys():
   768                  contents.append(name)
   769  
   770      return contents

   799  class ZlibArchiveWriter:
   800      """
   801      Writer for PyInstaller's PYZ (ZlibArchive) archive. The archive is used to store collected byte-compiled Python
   802      modules, as individually-compressed entries.
   803      """
   804      _PYZ_MAGIC_PATTERN = b'PYZ\0'
   805      _HEADER_LENGTH = 12 + 5
   806      _COMPRESSION_LEVEL = 6  # zlib compression level
   807  
   808      def __init__(self, filename, entries, code_dict=None):
   809          """
   810          filename
   811              Target filename of the archive.
   812          entries
   813              An iterable containing entries in the form of tuples: (name, src_path, typecode), where `name` is the name
   814              under which the resource is stored (e.g., python module name, without suffix), `src_path` is name of the
   815              file from which the resource is read, and `typecode` is the Analysis-level TOC typecode (`PYMODULE`).
   816          code_dict
   817              Optional code dictionary containing code objects for analyzed/collected python modules.
   818          """
   819          code_dict = code_dict or {}
   820  
   821          with open(filename, "wb") as fp:
   822              # Reserve space for the header.
   823              fp.write(b'\0' * self._HEADER_LENGTH)
   824  
   825              # Write entries' data and collect TOC entries
   826              toc = []
   827              for entry in entries:
   828                  toc_entry = self._write_entry(fp, entry, code_dict)
   829                  toc.append(toc_entry)
   830  
   831              # Write TOC
   832              toc_offset = fp.tell()
   833              toc_data = marshal.dumps(toc)
   834              fp.write(toc_data)
   835  
   836              # Write header:
   837              #  - PYZ magic pattern (4 bytes)
   838              #  - python bytecode magic pattern (4 bytes)
   839              #  - TOC offset (32-bit int, 4 bytes)
   840              #  - 4 unused bytes
   841              fp.seek(0, os.SEEK_SET)
   842  
   843              fp.write(self._PYZ_MAGIC_PATTERN)
   844              fp.write(BYTECODE_MAGIC)
   845              fp.write(struct.pack('!i', toc_offset))
   846  
   847      @classmethod
   848      def _write_entry(cls, fp, entry, code_dict):
   849          name, src_path, typecode = entry
   850          assert typecode in {'PYMODULE', 'PYMODULE-1', 'PYMODULE-2'}
   851  
   852          if src_path in {'-', None}:
   853              # PEP-420 namespace package; these do not have code objects, but we still need an entry in PYZ to inform our
   854              # run-time module finder/loader of the package's existence. So create a TOC entry for 0-byte data blob,
   855              # and write no data.
   856              return (name, (PYZ_ITEM_NSPKG, fp.tell(), 0))
   857  
   858          code_object = code_dict[name]
   859  
   860          src_basename, _ = os.path.splitext(os.path.basename(src_path))
   861          if src_basename == '__init__':
   862              typecode = PYZ_ITEM_PKG
   863              co_filename = os.path.join(*name.split('.'), '__init__.py')
   864          else:
   865              typecode = PYZ_ITEM_MODULE
   866              co_filename = os.path.join(*name.split('.')) + '.py'
   867  
   868          # Replace co_filename on code object with anonymized version without absolute path to the module.
   869          code_object = replace_filename_in_code_object(code_object, co_filename)
   870  
   871          # Serialize
   872          data = marshal.dumps(code_object)
   873  
   874          # First compress, then encrypt.
   875          obj = zlib.compress(data, cls._COMPRESSION_LEVEL)
   876  
   877          # Create TOC entry
   878          toc_entry = (name, (typecode, fp.tell(), len(obj)))
   879  
   880          # Write data blob
   881          fp.write(obj)
   882  
   883          return toc_entry

   886  class CArchiveWriter:
   887      """
   888      Writer for PyInstaller's CArchive (PKG) archive.
   889  
   890      This archive contains all files that are bundled within an executable; a PYZ (ZlibArchive), DLLs, Python C
   891      extensions, and other data files that are bundled in onefile mode.
   892  
   893      The archive can be read from either C (bootloader code at application's run-time) or Python (for debug purposes).
   894      """
   895      _COOKIE_MAGIC_PATTERN = b'MEI\014\013\012\013\016'
   896  
   897      # For cookie and TOC entry structure, see `PyInstaller.archive.readers.CArchiveReader`.
   898      _COOKIE_FORMAT = '!8sIIII64s'
   899      _COOKIE_LENGTH = struct.calcsize(_COOKIE_FORMAT)
   900  
   901      _TOC_ENTRY_FORMAT = '!IIIIBc'
   902      _TOC_ENTRY_LENGTH = struct.calcsize(_TOC_ENTRY_FORMAT)
   903  
   904      _COMPRESSION_LEVEL = 9  # zlib compression level
   905  
   906      def __init__(self, filename, entries, pylib_name):
   907          """
   908          filename
   909              Target filename of the archive.
   910          entries
   911              An iterable containing entries in the form of tuples: (dest_name, src_name, compress, typecode), where
   912              `dest_name` is the name under which the resource is stored in the archive (and name under which it is
   913              extracted at runtime), `src_name` is name of the file from which the resouce is read, `compress` is a
   914              boolean compression flag, and `typecode` is the Analysis-level TOC typecode.
   915          pylib_name
   916              Name of the python shared library.
   917          """
   918          self._collected_names = set()  # Track collected names for strict package mode.
   919  
   920          with open(filename, "wb") as fp:
   921              # Write entries' data and collect TOC entries
   922              toc = []
   923              for entry in entries:
   924                  toc_entry = self._write_entry(fp, entry)
   925                  toc.append(toc_entry)
   926  
   927              # Write TOC
   928              toc_offset = fp.tell()
   929              toc_data = self._serialize_toc(toc)
   930              toc_length = len(toc_data)
   931  
   932              fp.write(toc_data)
   933  
   934              # Write cookie
   935              archive_length = toc_offset + toc_length + self._COOKIE_LENGTH
   936              pyvers = sys.version_info[0] * 100 + sys.version_info[1]
   937              cookie_data = struct.pack(
   938                  self._COOKIE_FORMAT,
   939                  self._COOKIE_MAGIC_PATTERN,
   940                  archive_length,
   941                  toc_offset,
   942                  toc_length,
   943                  pyvers,
   944                  pylib_name.encode('ascii'),
   945              )
   946  
   947              fp.write(cookie_data)
   948  
   949      def _write_entry(self, fp, entry):
   950          dest_name, src_name, compress, typecode = entry
   951  
   952          # Write OPTION entries as-is, without normalizing them. This also exempts them from duplication check,
   953          # allowing them to be specified multiple times.
   954          if typecode == 'o':
   955              return self._write_blob(fp, b"", dest_name, typecode)
   956  
   957          # Ensure forward slashes in paths are on Windows converted to back slashes '\\', as on Windows the bootloader
   958          # works only with back slashes.
   959          dest_name = os.path.normpath(dest_name)
   960          if is_win and os.path.sep == '/':
   961              # When building under MSYS, the above path normalization uses Unix-style separators, so replace them
   962              # manually.
   963              dest_name = dest_name.replace(os.path.sep, '\\')
   964  
   965              # For symbolic link entries, also ensure that the symlink target path (stored in src_name) is using
   966              # Windows-style back slash separators.
   967              if typecode == 'n':
   968                  src_name = src_name.replace(os.path.sep, '\\')
   969  
   970          # Strict pack/collect mode: keep track of the destination names, and raise an error if we try to add a duplicate
   971          # (a file with same destination name, subject to OS case normalization rules).
   972          if strict_collect_mode:
   973              normalized_dest = None
   974              if typecode in {'s', 's1', 's2', 'm', 'M'}:
   975                  # Exempt python source scripts and modules from the check.
   976                  pass
   977              else:
   978                  # Everything else; normalize the case
   979                  normalized_dest = os.path.normcase(dest_name)
   980              # Check for existing entry, if applicable
   981              if normalized_dest:
   982                  if normalized_dest in self._collected_names:
   983                      raise ValueError(
   984                          f"Attempting to collect a duplicated file into CArchive: {normalized_dest} (type: {typecode})"
   985                      )
   986                  self._collected_names.add(normalized_dest)
   987  
   988          if typecode == 'd':
   989              # Dependency; merge src_name (= reference path prefix) and dest_name (= name) into single-string format that
   990              # is parsed by bootloader.
   991              return self._write_blob(fp, b"", f"{src_name}:{dest_name}", typecode)
   992          elif typecode in {'s', 's1', 's2'}:
   993              # If it is a source code file, compile it to a code object and marshal the object, so it can be unmarshalled
   994              # by the bootloader. For that, we need to know target optimization level, which is stored in typecode.
   995              optim_level = {'s': 0, 's1': 1, 's2': 2}[typecode]
   996              code = get_code_object(dest_name, src_name, optimize=optim_level)
   997              co_filename = dest_name + os.path.splitext(src_name)[1]  # Use dest name with suffix from source filename.
   998              code = replace_filename_in_code_object(code, co_filename)
   999              return self._write_blob(fp, marshal.dumps(code), dest_name, 's', compress=compress)
  1000          elif typecode in ('m', 'M'):
  1001              # Read the PYC file. We do not perform compilation here (in contrast to script files in the above branch),
  1002              # so typecode does not contain optimization level information.
  1003              with open(src_name, "rb") as in_fp:
  1004                  data = in_fp.read()
  1005              assert data[:4] == BYTECODE_MAGIC
  1006              # Skip the PYC header, load the code object.
  1007              code = marshal.loads(data[16:])
  1008              co_filename = dest_name + '.py'  # Use dest name with added .py suffix.
  1009              code = replace_filename_in_code_object(code, co_filename)
  1010              # These module entries are loaded and executed within the bootloader, which requires only the code
  1011              # object, without the PYC header.
  1012              return self._write_blob(fp, marshal.dumps(code), dest_name, typecode, compress=compress)
  1013          elif typecode == 'n':
  1014              # Symbolic link; store target name (as NULL-terminated string)
  1015              data = src_name.encode('utf-8') + b'\x00'
  1016              return self._write_blob(fp, data, dest_name, typecode, compress=compress)
  1017          else:
  1018              return self._write_file(fp, src_name, dest_name, typecode, compress=compress)
  1019  
  1020      def _write_blob(self, out_fp, blob: bytes, dest_name, typecode, compress=False):
  1021          """
  1022          Write the binary contents (**blob**) of a small file to the archive and return the corresponding CArchive TOC
  1023          entry.
  1024          """
  1025          data_offset = out_fp.tell()
  1026          data_length = len(blob)
  1027          if compress:
  1028              blob = zlib.compress(blob, level=self._COMPRESSION_LEVEL)
  1029          out_fp.write(blob)
  1030  
  1031          return (data_offset, len(blob), data_length, int(compress), typecode, dest_name)
  1032  
  1033      def _write_file(self, out_fp, src_name, dest_name, typecode, compress=False):
  1034          """
  1035          Stream copy a large file into the archive and return the corresponding CArchive TOC entry.
  1036          """
  1037          data_offset = out_fp.tell()
  1038          data_length = os.stat(src_name).st_size
  1039          with open(src_name, 'rb') as in_fp:
  1040              if compress:
  1041                  tmp_buffer = bytearray(16 * 1024)
  1042                  compressor = zlib.compressobj(self._COMPRESSION_LEVEL)
  1043                  while True:
  1044                      num_read = in_fp.readinto(tmp_buffer)
  1045                      if not num_read:
  1046                          break
  1047                      out_fp.write(compressor.compress(tmp_buffer[:num_read]))
  1048                  out_fp.write(compressor.flush())
  1049              else:
  1050                  shutil.copyfileobj(in_fp, out_fp)
  1051  
  1052          return (data_offset, out_fp.tell() - data_offset, data_length, int(compress), typecode, dest_name)
  1053  
  1054      @classmethod
  1055      def _serialize_toc(cls, toc):
  1056          serialized_toc = []
  1057          for toc_entry in toc:
  1058              data_offset, compressed_length, data_length, compress, typecode, name = toc_entry
  1059  
  1060              # Encode names as UTF-8. This should be safe as standard python modules only contain ASCII-characters (and
  1061              # standard shared libraries should have the same), and thus the C-code still can handle this correctly.
  1062              name = name.encode('utf-8')
  1063              name_length = len(name) + 1  # Add 1 for string-terminating zero byte.
  1064  
  1065              # Ensure TOC entries are aligned on 16-byte boundary, so they can be read by bootloader (C code) on
  1066              # platforms with strict data alignment requirements (for example linux on `armhf`/`armv7`, such as 32-bit
  1067              # Debian Buster on Raspberry Pi).
  1068              entry_length = cls._TOC_ENTRY_LENGTH + name_length
  1069              if entry_length % 16 != 0:
  1070                  padding_length = 16 - (entry_length % 16)
  1071                  name_length += padding_length
  1072  
  1073              # Serialize
  1074              serialized_entry = struct.pack(
  1075                  cls._TOC_ENTRY_FORMAT + f"{name_length}s",  # "Ns" format automatically pads the string with zero bytes.
  1076                  cls._TOC_ENTRY_LENGTH + name_length,
  1077                  data_offset,
  1078                  compressed_length,
  1079                  data_length,
  1080                  compress,
  1081                  typecode.encode('ascii'),
  1082                  name,
  1083              )
  1084              serialized_toc.append(serialized_entry)
  1085  
  1086          return b''.join(serialized_toc)

  1089  class SplashWriter:
  1090      """
  1091      Writer for the splash screen resources archive.
  1092  
  1093      The resulting archive is added as an entry into the CArchive with the typecode PKG_ITEM_SPLASH.
  1094      """
  1095      # This struct describes the splash resources as it will be in an buffer inside the bootloader. All necessary parts
  1096      # are bundled, the *_len and *_offset fields describe the data beyond this header definition.

  1098      # is null-terminated in order to easily iterate over this list from within C.

  1102      #     char tcl_libname[16];

  1110      #     uint32_t image_offset;

  1113      #     uint32_t requirements_offset;

  1122          """
  1123          Writer for splash screen resources that are bundled into the CArchive as a single archive/entry.
  1124  
  1125          :param filename: The filename of the archive to create
  1126          :param name_list: List of filenames for the requirements array
  1127          :param str tcl_libname: Name of the tcl shared library file
  1128          :param str tk_libname: Name of the tk shared library file
  1129          :param str tklib: Root of tk library (e.g. tk/)

  1131          :param str script: The tcl/tk script to execute to create the screen.

  1133  
  1134          # Ensure forward slashes in dependency names are on Windows converted to back slashes '\\', as on Windows the
  1135          # bootloader works only with back slashes.
  1136          def _normalize_filename(filename):
  1137              filename = os.path.normpath(filename)
  1138              if is_win and os.path.sep == '/':
  1139                  # When building under MSYS, the above path normalization uses Unix-style separators, so replace them
  1140                  # manually.
  1141                  filename = filename.replace(os.path.sep, '\\')
  1142              return filename

  1144          name_list = [_normalize_filename(name) for name in name_list]
  1145  
  1146          with open(filename, "wb") as fp:
  1147              # Reserve space for the header.
  1148              fp.write(b'\0' * self._HEADER_LENGTH)
  1149  
  1150              # Serialize the requirements list. This list (more an array) contains the names of all files the bootloader
  1151              # needs to extract before the splash screen can be started. The implementation terminates every name with a
  1152              # null-byte, that keeps the list short memory wise and makes it iterable from C.
  1153              requirements_len = 0
  1154              requirements_offset = fp.tell()
  1155              for name in name_list:
  1156                  name = name.encode('utf-8') + b'\0'
  1157                  fp.write(name)
  1158                  requirements_len += len(name)
  1159  
  1160              # Write splash script
  1161              script_offset = fp.tell()
  1162              script_len = len(script)
  1163              fp.write(script.encode("utf-8"))
  1164  
  1165              # Write splash image. If image is a bytes buffer, it is written directly into the archive. Otherwise, it
  1166              # is assumed to be a path and the file is copied into the archive.
  1167              image_offset = fp.tell()
  1168              if isinstance(image, bytes):
  1169                  # Image was converted by PIL/Pillow and is already in buffer
  1170                  image_len = len(image)
  1171                  fp.write(image)
  1172              else:
  1173                  # Read image into buffer
  1174                  with open(image, 'rb') as image_fp:
  1175                      image_data = image_fp.read()
  1176                  image_len = len(image_data)
  1177                  fp.write(image_data)
  1178                  del image_data
  1179  
  1180              # The following strings are written to 16-character fields with zero-padding, which means that we need to
  1181              # ensure that their length is strictly below 16 characters (if it were exactly 16, the field would have no
  1182              # terminating NULL character!).
  1183              def _encode_str(value, field_name, limit):
  1184                  enc_value = value.encode("utf-8")

  1186                      raise ValueError(
  1187                          f"Length of the encoded field {field_name!r} ({len(enc_value)}) is greater or equal to the "
  1188                          f"limit of {limit} characters!"
  1189                      )
  1190  
  1191                  return enc_value
  1192  
  1193              # Write header
  1194              header_data = struct.pack(
  1195                  self._HEADER_FORMAT,
  1196                  _encode_str(tcl_libname, 'tcl_libname', 32),
  1197                  _encode_str(tk_libname, 'tk_libname', 32),
  1198                  _encode_str(tklib, 'tklib', 16),
  1199                  script_len,
  1200                  script_offset,
  1201                  image_len,
  1202                  image_offset,
  1203                  requirements_len,
  1204                  requirements_offset,
  1205              )
  1206  
  1207              fp.seek(0, os.SEEK_SET)
  1208              fp.write(header_data)
